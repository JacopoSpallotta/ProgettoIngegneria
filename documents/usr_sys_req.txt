Insulin_pump{
	sys_req{
		functional{
			Il livello di glucosio non deve scendere sotto i 50mg/dL (ipoglicemia);
			Il livello di glucosio non deve salire sopra i x mg/dL (iperglicemia);
			Il livello di glucosio dovrebbere rimanere il più viciono possibilea 100mg/dL;
			Il sistema garantisce il rilascio di una dose minima di insulina anche in caso 
				di fault di sistema;
			Il sistema analizza il trend di glicemia in salita o in discesa nel sangue
			    Con glicemia nei limiti di sicurezza:
			        In caso di livelli stabili o in dicesa di glucosiop:
			            La pompa non rilascia insulina.
                    In caso aumento di glucosio ma tasso di crescita in discesa:
                        La pompa non rilascia insulina.
                    In caso aumento di glucosio ma tasso di crescita in ascesa:
                        Calcola dose di insulina appropriata.
                    In caso di trend stabile:
                        La pompa non rilascia insulina.

				Con glicemia alta:
				    In caso di aumento glucosio:
				        La pompa rilascia una dose appropriata.
				    In caso di livello glucosio stabile:
				        La pompa rilascia una dose minima
                    In caso di diminuzione di glucosio ma tasso di decremento in ascesa:
                        La pompa non rilascia insulina.
                    In caso di diminuzione di glucosio con tasso di decremento in discesa:
                        La pompa rilascia una dose minima.
			
		}
		non_functional{
			La dose di insulina giornalmente deve essere limitata ad un massimo di x/unit;

		}
	}


	usr_req{
		functional{
			Garantire un livello di glucosio nel sangue introno ai 100mg/dL;
			Informare l'utente in caso di livelli troppo elevati o troppo bassi di glucosio;
			La pompa dell'insulina avverte l'utente che la riserva sta per esaurirsi;
			
		}
		non_functional{
			Minimizzare il quantitativo di insulina ignettata giornalmente.
			Massimizzare il tempo fra un controllo e il successivo, mantendosi fra 5 e 10 minuti.

		}
	}
}

				dello stato del paziente:
				    1. Livello del glucosio nel sangue (G - mg/dL)

				    4. Produzione endogena di glucosio (EGP - adimensinale)
				    5. Utilizzo dipendente di insulina (UID - mg/kg/min)
				    6. Escrezione renale (E - mg/kg/min)
				    7. Secrezione di Insulina e C-peptide (ISR - pmol/min)

		}

		}
	}
